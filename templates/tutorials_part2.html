num1 = int(input("Enter the first number: "))
num2 = int(input("Enter the second number: "))

print("Choose an Operation: +, -, *, / ")

operation = input()
if operation == "+":
    print("The answer is: ", num1 + num2)
elif operation == "-":
    print("The result is: ", num1 - num2)
elif operation == "*":
    print("The answer is: ", num1 * num2)
elif operation == "/":
    print("The answer is: ", num1 / num2) 
else:
    print("Invalid operation")
